Donut with legend and click events

{
    _: TK.Donut,
    Width: 500,
    Height: 200,
    Values: [
        { Name: "Value A", Color: "#F90", Value: 1 },
        { Name: "Value B", Color: "#93C", Value: 2 },
        { Name: "Value C", Color: "#09F", Value: 2 }
    ],    
    EnableLegend: true,
    LabelStyle: 1,
    Click: function(value) {
        alert('Clicked on ' + value.Name);
    }
}
